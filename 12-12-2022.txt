-----------------SCALAR FUNCTION (return single value)------------------------------
--creating a function

create function Scalar_pass_fail(@num int)
returns char 
as
begin
 declare @pass_Fail char
 if (@num >280)
   set @pass_Fail='A'
 else if (@num <280 and @num>250)
   set @pass_Fail='B'
 else if (@num <250 and @num>200)
   set @pass_Fail='C'
 else 
   set @pass_Fail='F'
 return @pass_Fail
end
select * , dbo.Scalar_pass_fail(total) as grade from Education_detail

select * , dbo.Scalar_pass_fail(total) as grade from Education_detail where stud_Roll_NO>=10

select Education_detail.stud_Roll_NO ,Students.stud_Name,Education_detail.subject1,Education_detail.subject2,
Education_detail.subject3,Education_detail.total,Education_detail.average, dbo.paas_Fail(total) as grade
from Education_detail 
inner join Students on Students.stud_Roll_NO=Education_detail.stud_Roll_NO
